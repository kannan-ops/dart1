import 'dart:math';

void main() {
  int number = 153;
  int original = number;
  int sum = 0;

  while (number > 0) {
    int digit = number % 10;
    sum += pow(digit, 3).toInt();  // cube panrom
    number ~/= 10; // next digit
  }

  if (sum == original) {
    print("$original is an Armstrong number");
  } else {
    print("$original is not an Armstrong number");
  }
}
