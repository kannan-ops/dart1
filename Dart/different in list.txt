void main() {
  List<int> num = [1, 2, 3, 4, 5,6,7,8,9];
  List<int> different = [];

  for (int i = 0; i < num.length; i++) {
    if (!different.contains(num[i])) {
      different.add(num[i]);
    }
  }

  int result = different.length - 1;  

  print("different in list: $result");
}
