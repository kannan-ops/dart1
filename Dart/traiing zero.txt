void main() {
  int num = 5;
  int factorial = 1;

  for (int i = 1; i <= num; i++) {
    factorial *= i;
  }

  print("Factorial of $num is $factorial");

 
  int count = 0;
  int temp = factorial;

  while (temp % 10 == 0) {   
    count++;
    temp ~/= 10;
  }

  print("Number of trailing zeros = $count");
}
